// Build the inner HTML for the screenshot grid for a given product.
// Exposed so the renderSide() template literal can call it inline.
function renderScreenshotWidget(p) {
  const shots = Array.isArray(p && p.screenshots) ? p.screenshots : [];
  if (!shots.length) {
    return `<div class="gallery-empty">no screenshots &mdash; upload or drop up to 6 images</div>`;
  }
  return shots.map((path, i) => {
    const staged = S.stagedImages && S.stagedImages.get(path);
    const src = staged ? staged.dataUrl : path;
    return `
      <div class="gallery-thumb${staged ? ' staged' : ''}" draggable="true" data-i="${i}">
        <img src="${escapeHtml(src)}" alt="screenshot ${i + 1}" onerror="this.style.opacity=0.3" />
        <span class="slot-badge">#${i + 1}</span>
        ${i === 0 ? '<span class="primary-badge">PRIMARY</span>' : ''}
        <button type="button" class="remove-x" title="Remove" onclick="removeScreenshot(${i})">&times;</button>
      </div>`;
  }).join('');
}

// Redraw just the gallery area (cheaper than full renderSide()).
function refreshGalleryGrid() {
  const p = S.products.find(x => x.id === S.selectedId);
  if (!p) return;
  const grid = document.getElementById('gallery-grid');
  if (!grid) return;
  grid.innerHTML = renderScreenshotWidget(p);
  wireGalleryDnD();
  refreshScreenshotCount();
}

function refreshScreenshotCount() {
  const p = S.products.find(x => x.id === S.selectedId);
  if (!p) return;
  const el = document.getElementById('ss-count');
  if (el) el.textContent = (Array.isArray(p.screenshots) ? p.screenshots.length : 0);
}

// Wire native HTML5 drag-and-drop on each thumb for reorder.
function wireGalleryDnD() {
  const grid = document.getElementById('gallery-grid');
  if (!grid) return;
  const thumbs = grid.querySelectorAll('.gallery-thumb');
  let fromIdx = null;
  thumbs.forEach(th => {
    th.addEventListener('dragstart', (e) => {
      fromIdx = parseInt(th.dataset.i, 10);
      th.classList.add('dragging');
      // Use a transparent payload so it's treated as same-origin move, not a file drop
      try { e.dataTransfer.setData('text/plain', 'gallery:' + fromIdx); } catch (_) {}
      if (e.dataTransfer) e.dataTransfer.effectAllowed = 'move';
    });
    th.addEventListener('dragend', () => {
      th.classList.remove('dragging');
      grid.querySelectorAll('.gallery-thumb').forEach(t => t.classList.remove('drop-target'));
      fromIdx = null;
    });
    th.addEventListener('dragover', (e) => {
      // Only act on internal gallery drags (not file drops onto the thumb)
      if (fromIdx == null) return;
      e.preventDefault();
      if (e.dataTransfer) e.dataTransfer.dropEffect = 'move';
      grid.querySelectorAll('.gallery-thumb').forEach(t => t.classList.remove('drop-target'));
      th.classList.add('drop-target');
    });
    th.addEventListener('dragleave', () => {
      th.classList.remove('drop-target');
    });
    th.addEventListener('drop', (e) => {
      if (fromIdx == null) return;
      e.preventDefault();
      e.stopPropagation();
      const toIdx = parseInt(th.dataset.i, 10);
      if (fromIdx !== toIdx) reorderScreenshots(fromIdx, toIdx);
    });
  });
}

// Stage one uploaded image at the next available slot (or targetIndex if given).
function stageScreenshotUpload(file, targetIndex) {
  const p = S.products.find(x => x.id === S.selectedId);
  if (!p) return;
  if (!/^image\//i.test(file.type)) { toast('Not an image file', 'err'); return; }
  if (file.size > 4 * 1024 * 1024) { toast('Image too large (>4MB). Compress and retry.', 'err'); return; }

  p.screenshots = Array.isArray(p.screenshots) ? p.screenshots : [];
  if (p.screenshots.length >= 6 && targetIndex == null) {
    toast('Max 6 screenshots per product', 'err');
    return;
  }

  const reader = new FileReader();
  reader.onload = () => {
    const dataUrl = reader.result;
    const comma = dataUrl.indexOf(',');
    const base64Content = dataUrl.slice(comma + 1);
    const ext = (file.name.split('.').pop() || 'jpg').toLowerCase().replace(/[^a-z0-9]/g, '') || 'jpg';
    const slug = p.slug || slugify(p.name || p.id || 'product');
    const slotIndex = (typeof targetIndex === 'number') ? targetIndex : p.screenshots.length;
    const humanIndex = slotIndex + 1; // 1-based in filename for clarity
    const targetPath = 'assets/products/' + slug + '-ss-' + humanIndex + '.' + ext;
    const publicPath = '/' + targetPath;

    S.stagedImages.set(publicPath, {
      repoPath: targetPath,
      filename: file.name,
      base64: base64Content,
      dataUrl: dataUrl
    });

    snapshot();
    // If target was given and a slot existed, replace; otherwise append
    if (typeof targetIndex === 'number' && p.screenshots[targetIndex] != null) {
      // clear stale staged entry if present for the old path
      const oldPath = p.screenshots[targetIndex];
      if (oldPath && S.stagedImages.has(oldPath) && oldPath !== publicPath) {
        S.stagedImages.delete(oldPath);
      }
      p.screenshots[targetIndex] = publicPath;
    } else {
      p.screenshots.push(publicPath);
    }

    markDirty(p.id);
    refreshGalleryGrid();
    const hint = document.getElementById('gallery-hint');
    if (hint) {
      hint.textContent = `Staged ${file.name} to slot #${humanIndex}. Commits on next Publish.`;
      hint.classList.add('staged');
    }
    toast('Screenshot staged', 'ok');
  };
  reader.readAsDataURL(file);
}

function removeScreenshot(index) {
  const p = S.products.find(x => x.id === S.selectedId);
  if (!p || !Array.isArray(p.screenshots)) return;
  if (index < 0 || index >= p.screenshots.length) return;
  snapshot();
  const removed = p.screenshots.splice(index, 1)[0];
  // If it was staged-but-not-yet-committed, drop the staged blob to free memory
  // and prevent the publish flow from pushing an orphan file.
  if (removed && S.stagedImages && S.stagedImages.has(removed)) {
    S.stagedImages.delete(removed);
  }
  markDirty(p.id);
  refreshGalleryGrid();
  toast('Screenshot removed', 'ok');
}

function reorderScreenshots(from, to) {
  const p = S.products.find(x => x.id === S.selectedId);
  if (!p || !Array.isArray(p.screenshots)) return;
  if (from === to || from < 0 || to < 0 || from >= p.screenshots.length || to >= p.screenshots.length) return;
  snapshot();
  const [moved] = p.screenshots.splice(from, 1);
  p.screenshots.splice(to, 0, moved);
  markDirty(p.id);
  refreshGalleryGrid();
  toast('Reordered', 'ok');
}

  (function () {
    var lb = document.getElementById('ss-lightbox');
    if (!lb) return;
    var img = document.getElementById('ss-lb-img');
    var closeBtn = lb.querySelector('.ss-lb-close');
    var prevBtn = lb.querySelector('.ss-lb-prev');
    var nextBtn = lb.querySelector('.ss-lb-next');
    var triggers = Array.prototype.slice.call(document.querySelectorAll('[data-ss-src]'));
    // Dedupe sources in document order (primary + thumbs share indices)
    var sources = [];
    triggers.forEach(function (t) {
      var s = t.getAttribute('data-ss-src');
      if (s && sources.indexOf(s) === -1) sources.push(s);
    });
    var idx = 0;
    function open(i) {
      if (!sources.length) return;
      idx = ((i % sources.length) + sources.length) % sources.length;
      img.src = sources[idx];
      lb.classList.add('is-open');
      document.body.style.overflow = 'hidden';
    }
    function close() {
      lb.classList.remove('is-open');
      img.src = '';
      document.body.style.overflow = '';
    }
    triggers.forEach(function (t) {
      t.addEventListener('click', function (e) {
        e.preventDefault();
        var src = t.getAttribute('data-ss-src');
        var i = sources.indexOf(src);
        open(i < 0 ? 0 : i);
      });
    });
    closeBtn.addEventListener('click', close);
    prevBtn.addEventListener('click', function () { open(idx - 1); });
    nextBtn.addEventListener('click', function () { open(idx + 1); });
    lb.addEventListener('click', function (e) { if (e.target === lb) close(); });
    document.addEventListener('keydown', function (e) {
      if (!lb.classList.contains('is-open')) return;
      if (e.key === 'Escape') close();
      else if (e.key === 'ArrowLeft') open(idx - 1);
      else if (e.key === 'ArrowRight') open(idx + 1);
    });
  })();
</script>
