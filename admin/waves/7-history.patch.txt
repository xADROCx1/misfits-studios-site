/* Toggle setup card vs body based on whether a PAT is stored. */
function syncHistoryView() {
  const hasPAT = !!getStoredPAT();
  document.getElementById('history-setup').style.display = hasPAT ? 'none' : '';
  document.getElementById('history-body').style.display  = hasPAT ? '' : 'none';
  if (hasPAT && HIST.commits.length === 0 && !HIST.loading) {
    loadHistory();
  } else if (hasPAT) {
    renderHistoryTable();
  }
}

/* Fetch the last 30 commits that touched products.json. */
async function loadHistory() {
  const token = getStoredPAT();
  if (!token) { syncHistoryView(); return; }

  HIST.loading = true;
  HIST.error = '';
  renderHistorySkeleton();

  const btn = document.getElementById('history-refresh-btn');
  if (btn) { btn.disabled = true; btn.textContent = '↻ Loading...'; }

  try {
    const url = `https://api.github.com/repos/${HIST.repo}/commits`
              + `?path=${encodeURIComponent(HIST.path)}&per_page=30&sha=${HIST.branch}`;
    const res = await fetch(url, {
      headers: {
        'Authorization': 'token ' + token,
        'Accept': 'application/vnd.github+json',
      },
    });
    if (res.status === 401 || res.status === 403) {
      const j = await res.json().catch(() => ({}));
      const rate = res.headers.get('X-RateLimit-Remaining');
      if (rate === '0') throw new Error('GitHub rate-limit hit. Wait a few minutes or use a different token.');
      throw new Error('GitHub rejected the token (' + res.status + '). ' + (j.message || 'Check scope/expiry.'));
    }
    if (!res.ok) throw new Error('GitHub ' + res.status + ' ' + res.statusText);
    const data = await res.json();
    HIST.commits = data.map(c => ({
      sha: c.sha,
      message: (c.commit && c.commit.message) || '',
      author: (c.commit && c.commit.author && c.commit.author.name)
            || (c.author && c.author.login) || 'unknown',
      date: (c.commit && c.commit.author && c.commit.author.date) || '',
      expanded: false,
      details: null,
    }));
    HIST.loading = false;
    renderHistoryTable();
  } catch (e) {
    HIST.loading = false;
    HIST.error = (e && e.message) || String(e);
    renderHistoryTable();
    toast('History: ' + HIST.error, 'err');
  } finally {
    if (btn) { btn.disabled = false; btn.textContent = '↻ Refresh'; }
  }
}

function renderHistorySkeleton() {
  const tbody = document.getElementById('history-rows');
  if (!tbody) return;
  let html = '';
  for (let i = 0; i < 8; i++) {
    html += '<tr><td><div class="history-skel" style="width:70px"></div></td>'
         +  '<td><div class="history-skel" style="width:80%"></div></td>'
         +  '<td><div class="history-skel" style="width:90px"></div></td>'
         +  '<td><div class="history-skel" style="width:60px"></div></td>'
         +  '<td><div class="history-skel" style="width:180px;margin-left:auto"></div></td></tr>';
  }
  tbody.innerHTML = html;
}

function renderHistoryTable() {
  const tbody = document.getElementById('history-rows');
  if (!tbody) return;
  if (HIST.error) {
    tbody.innerHTML = `<tr><td colspan="5" class="history-error">${escapeHtml(HIST.error)}<br/><span style="color:var(--muted);font-size:11px;">Click Refresh to retry.</span></td></tr>`;
    return;
  }
  if (HIST.commits.length === 0) {
    tbody.innerHTML = `<tr><td colspan="5" class="history-empty">No commits found that touched ${escapeHtml(HIST.path)}.</td></tr>`;
    return;
  }

  const metaEl = document.getElementById('history-meta');
  if (metaEl) metaEl.textContent = HIST.commits.length + ' commits that touched ' + HIST.path;

  let html = '';
  HIST.commits.forEach((c, idx) => {
    const short = c.sha.slice(0, 7);
    const firstLine = (c.message || '').split('\n')[0];
    const msgTrunc = firstLine.length > 80 ? firstLine.slice(0, 77) + '…' : firstLine;
    const isHead = idx === 0;
    html += `<tr class="history-row" data-sha="${escapeHtml(c.sha)}">
      <td><span class="history-sha${isHead ? ' head' : ''}" title="${escapeHtml(c.sha)}">${short}</span></td>
      <td><div class="history-msg" title="${escapeHtml(firstLine)}">${escapeHtml(msgTrunc)}</div></td>
      <td><span class="history-author">${escapeHtml(c.author)}</span></td>
      <td><span class="history-when" title="${escapeHtml(c.date)}">${escapeHtml(relativeTime(c.date))}</span></td>
      <td class="num"><div class="history-actions">
        <button class="btn" onclick="viewDiff('${c.sha}')">${c.expanded ? '▴ Hide diff' : '▾ View diff'}</button>
        <button class="btn btn-danger" onclick="restoreVersion('${c.sha}')" ${isHead ? 'disabled title="This IS the current live version"' : ''}>&#8630; Restore</button>
      </div></td>
    </tr>`;
    if (c.expanded) {
      html += `<tr class="history-diff-row"><td colspan="5">${renderDiffBox(c)}</td></tr>`;
    }
  });
  tbody.innerHTML = html;
}

/* Toggle-expand a row and lazy-fetch the per-commit files[] payload. */
async function viewDiff(sha) {
  const c = HIST.commits.find(x => x.sha === sha);
  if (!c) return;
  c.expanded = !c.expanded;
  renderHistoryTable();
  if (!c.expanded || c.details) return;

  const token = getStoredPAT();
  if (!token) { toast('No GitHub token saved', 'err'); return; }

  try {
    const res = await fetch(
      `https://api.github.com/repos/${HIST.repo}/commits/${sha}`,
      { headers: { 'Authorization': 'token ' + token, 'Accept': 'application/vnd.github+json' } }
    );
    if (!res.ok) throw new Error('GitHub ' + res.status);
    const data = await res.json();
    c.details = {
      stats: data.stats || { additions: 0, deletions: 0, total: 0 },
      files: (data.files || []).map(f => ({
        filename: f.filename,
        status: f.status,
        additions: f.additions || 0,
        deletions: f.deletions || 0,
      })),
    };
    // If Compare-to-current is on and this is the selected commit, also fetch content for the key diff.
    if (HIST.compareMode) await hydrateCompareFor(sha);
    renderHistoryTable();
  } catch (e) {
    toast('Diff fetch failed: ' + e.message, 'err');
    c.expanded = false;
    renderHistoryTable();
  }
}

function renderDiffBox(c) {
  if (!c.details) {
    return `<div class="history-diff-box"><div class="history-diff-title">Loading diff…</div><div class="history-skel" style="width:40%"></div></div>`;
  }
  const s = c.details.stats;
  let files = '';
  c.details.files.forEach(f => {
    files += `<div class="history-diff-file">
      <span class="history-diff-fn">${escapeHtml(f.filename)} <span style="color:var(--muted)">(${escapeHtml(f.status)})</span></span>
      <span class="history-diff-add">+${f.additions}</span>
      <span class="history-diff-del">-${f.deletions}</span>
    </div>`;
  });
  let compare = '';
  if (HIST.compareMode && c.compare) {
    compare = `<div class="history-compare">
      <div class="history-compare-title">Top-level key diff vs your current editor state</div>
      <div class="history-compare-row">
        <span class="add">Added keys: <b>${c.compare.added.length}</b>${c.compare.added.length ? ' · ' + c.compare.added.slice(0, 5).map(escapeHtml).join(', ') + (c.compare.added.length > 5 ? '…' : '') : ''}</span>
        <span class="del">Removed keys: <b>${c.compare.removed.length}</b>${c.compare.removed.length ? ' · ' + c.compare.removed.slice(0, 5).map(escapeHtml).join(', ') + (c.compare.removed.length > 5 ? '…' : '') : ''}</span>
        <span class="mod">Modified keys: <b>${c.compare.modified.length}</b>${c.compare.modified.length ? ' · ' + c.compare.modified.slice(0, 5).map(escapeHtml).join(', ') + (c.compare.modified.length > 5 ? '…' : '') : ''}</span>
      </div>
    </div>`;
  } else if (HIST.compareMode && c.compareError) {
    compare = `<div class="history-compare"><div class="history-compare-title" style="color:var(--danger)">Compare failed</div><div style="color:var(--muted)">${escapeHtml(c.compareError)}</div></div>`;
  }
  return `<div class="history-diff-box">
    <div class="history-diff-title">Commit totals · +${s.additions} −${s.deletions} across ${c.details.files.length} file(s)</div>
    ${files}
    ${compare}
  </div>`;
}

/* "Compare to current" — fetch this commit's products.json and diff top-level product
   ids + manifest keys against the current editor S.raw / S.products. Kept intentionally
   simple: added/removed/modified product-id counts + raw top-level key counts. */
async function hydrateCompareFor(sha) {
  const c = HIST.commits.find(x => x.sha === sha);
  if (!c) return;
  if (c.compare || c.compareError) return;
  const token = getStoredPAT();
  if (!token) return;
  try {
    const contentRes = await fetch(
      `https://api.github.com/repos/${HIST.repo}/contents/${encodeURIComponent(HIST.path)}?ref=${sha}`,
      { headers: { 'Authorization': 'token ' + token, 'Accept': 'application/vnd.github+json' } }
    );
    if (!contentRes.ok) throw new Error('GitHub ' + contentRes.status);
    const ct = await contentRes.json();
    const past = JSON.parse(atob((ct.content || '').replace(/\n/g, '')));
    c.compare = diffAgainstCurrent(past);
  } catch (e) {
    c.compareError = e.message || String(e);
  }
}

/* Simple per-product-id diff + top-level key diff. Intentionally shallow. */
function diffAgainstCurrent(pastRaw) {
  const cur = S.raw || { products: S.products || [] };
  const curIds = new Set((cur.products || []).map(p => p.id));
  const pastIds = new Set((pastRaw.products || []).map(p => p.id));
  const added = [];     // in past, not in current  (would be restored)
  const removed = [];   // in current, not in past  (would disappear on restore)
  const modified = [];  // same id, different content
  pastIds.forEach(id => { if (!curIds.has(id)) added.push(id); });
  curIds.forEach(id => { if (!pastIds.has(id)) removed.push(id); });
  const curById = new Map((cur.products || []).map(p => [p.id, JSON.stringify(p)]));
  (pastRaw.products || []).forEach(p => {
    if (curIds.has(p.id) && curById.get(p.id) !== JSON.stringify(p)) modified.push(p.id);
  });
  return { added, removed, modified };
}

function onHistoryCompareToggle() {
  HIST.compareMode = document.getElementById('history-compare-toggle').checked;
  // Re-hydrate for any already-expanded rows.
  const expanded = HIST.commits.filter(c => c.expanded);
  Promise.all(expanded.map(c => hydrateCompareFor(c.sha))).then(() => renderHistoryTable());
  renderHistoryTable();
}

/* Fetch a past version of products.json and load it into the editor as dirty state.
   We deliberately DO NOT call markAllClean — the user needs to Publish to make it live. */
async function restoreVersion(sha) {
  const c = HIST.commits.find(x => x.sha === sha);
  const short = sha.slice(0, 7);
  const msg = (c && c.message.split('\n')[0]) || '';

  const ok = confirm(
    `Restore products.json from commit ${short}?\n\n` +
    (msg ? `"${msg}"\n\n` : '') +
    `Your current unsaved changes will be lost. This loads the old content into the editor — ` +
    `you still need to Publish to make it live.`
  );
  if (!ok) return;

  const token = getStoredPAT();
  if (!token) { toast('No GitHub token saved — open Publish to save one', 'err'); return; }

  try {
    const res = await fetch(
      `https://api.github.com/repos/${HIST.repo}/contents/${encodeURIComponent(HIST.path)}?ref=${sha}`,
      { headers: { 'Authorization': 'token ' + token, 'Accept': 'application/vnd.github+json' } }
    );
    if (!res.ok) throw new Error('GitHub ' + res.status + ' ' + res.statusText);
    const data = await res.json();
    if (!data.content) throw new Error('No content returned for that commit');
    // GitHub returns base64 with embedded newlines.
    const decoded = atob(data.content.replace(/\n/g, ''));
    let parsed;
    try { parsed = JSON.parse(decoded); }
    catch (pe) { throw new Error('That version is not valid JSON: ' + pe.message); }
    if (!parsed.products || !Array.isArray(parsed.products)) {
      throw new Error('That version has no "products" array — refusing to load.');
    }

    // Swap editor state to the restored snapshot.
    S.raw = parsed;
    S.products = parsed.products;
    S.selectedId = null;

    // Re-render every view with the restored data.
    rerenderAll();

    // DO NOT markAllClean — we want the user to see "unsaved changes" and Publish.
    // Instead, flag every product + the manifest dirty so the badge + grid tint both fire.
    if (!S.dirty) S.dirty = new Set();
    S.dirty.clear();
    S.products.forEach(p => S.dirty.add(p.id));
    S.dirty.add('__manifest__');
    updateDirtyBadge();
    renderGrid();

    toast('Restored version ' + short + ' — review and hit Publish to make it live', 'ok');
  } catch (e) {
    toast('Restore failed: ' + (e.message || e), 'err');
  }
}

/* Relative-time formatter — small, no-deps. */
function relativeTime(iso) {
  if (!iso) return '—';
  const then = new Date(iso).getTime();
  if (isNaN(then)) return '—';
  const diff = Math.max(0, Date.now() - then);
  const s = Math.floor(diff / 1000);
  if (s < 60)   return s + 's ago';
  const m = Math.floor(s / 60);
  if (m < 60)   return m + 'm ago';
  const h = Math.floor(m / 60);
  if (h < 24)   return h + 'h ago';
  const d = Math.floor(h / 24);
  if (d < 7)    return d + 'd ago';
  if (d < 30)   return Math.floor(d / 7) + 'w ago';
  if (d < 365)  return Math.floor(d / 30) + 'mo ago';
  return Math.floor(d / 365) + 'y ago';
}
