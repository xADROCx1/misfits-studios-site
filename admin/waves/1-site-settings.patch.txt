/* ---------- View switching ---------- */
function showSettingsView() {
  document.getElementById('grid').style.display = 'none';
  const fp = document.getElementById('featured-panel'); if (fp) fp.style.display = 'none';
  const sp = document.getElementById('sales-panel');    if (sp) sp.style.display = 'none';
  document.getElementById('settings-panel').style.display = '';
  document.getElementById('canvas-title').textContent = 'Site Settings';
  document.getElementById('canvas-sub').textContent = 'Brand · Pricing · Coming Soon · Announcement';
  renderSettingsPanel();
}

/* ---------- Ensure schema exists (defensive — products.json already has these) ---------- */
function settingsEnsureSchema() {
  if (!S.raw) return;
  if (!S.raw.store || typeof S.raw.store !== 'object') S.raw.store = {};
  if (!S.raw.pricing_tiers || typeof S.raw.pricing_tiers !== 'object') S.raw.pricing_tiers = {};
  if (!Array.isArray(S.raw.coming_soon)) S.raw.coming_soon = [];
  if (!S.raw.announcement || typeof S.raw.announcement !== 'object') {
    S.raw.announcement = { enabled: false, text: '', link: '', link_label: '', style: 'neon' };
  }
}

/* ---------- Snapshot string used for dirty detection by host ---------- */
function settingsSnapshotString() {
  if (!S.raw) return '';
  return JSON.stringify({
    store: S.raw.store || {},
    pricing_tiers: S.raw.pricing_tiers || {},
    coming_soon: S.raw.coming_soon || [],
    announcement: S.raw.announcement || {}
  });
}

/* ---------- Master render ---------- */
function renderSettingsPanel() {
  settingsEnsureSchema();
  settingsRenderBrand();
  settingsRenderTiers();
  settingsRenderComingSoon();
  settingsRenderAnnouncement();
}

/* ---------- Collapse / expand all ---------- */
function settingsToggleAll(open) {
  document.querySelectorAll('#settings-panel .settings-section').forEach(d => {
    d.open = !!open;
  });
}

/* ---------- Section 1: Brand & Contact ---------- */
function settingsRenderBrand() {
  const s = S.raw.store || {};
  const set = (id, v) => { const el = document.getElementById(id); if (el) el.value = v == null ? '' : v; };
  set('set-store-name',    s.name);
  set('set-store-tagline', s.tagline);
  set('set-store-email',   s.contact_email);
  set('set-store-discord', s.discord_url);
  set('set-store-twitter', s.twitter_url);
  set('set-store-github',  s.github_url);
}
function settingsOnStoreField(key, value) {
  settingsEnsureSchema();
  if (typeof snapshot === 'function') snapshot();
  S.raw.store[key] = value;
  S.dirty.add('__manifest__');
  updateDirtyBadge();
  // No grid re-render needed — nothing in the product grid depends on store.*
}

/* ---------- Section 2: Pricing Tiers ---------- */
const SETTINGS_TIER_ORDER = ['micro', 'small', 'medium', 'large', 'pro', 'flagship', 'bundle'];
function settingsRenderTiers() {
  settingsEnsureSchema();

function settingsAttachComingSoonDnD() {
  const rows = document.querySelectorAll('#settings-cs-list .settings-cs-row');
  rows.forEach(row => {
    row.addEventListener('dragstart', (e) => {
      e.dataTransfer.setData('text/plain', row.dataset.idx);
      e.dataTransfer.effectAllowed = 'move';
      row.classList.add('dragging');
    });
    row.addEventListener('dragend', () => {
      row.classList.remove('dragging');
      rows.forEach(r => { r.classList.remove('drop-above'); r.classList.remove('drop-below'); });
    });
    row.addEventListener('dragover', (e) => {
      e.preventDefault();
      e.dataTransfer.dropEffect = 'move';
      const rect = row.getBoundingClientRect();
      const below = (e.clientY - rect.top) > rect.height / 2;
      rows.forEach(r => { r.classList.remove('drop-above'); r.classList.remove('drop-below'); });
      row.classList.add(below ? 'drop-below' : 'drop-above');
    });
    row.addEventListener('drop', (e) => {
      e.preventDefault();
      const fromIdx = Number(e.dataTransfer.getData('text/plain'));
      const toIdx   = Number(row.dataset.idx);
      rows.forEach(r => { r.classList.remove('drop-above'); r.classList.remove('drop-below'); });
      if (Number.isNaN(fromIdx) || Number.isNaN(toIdx) || fromIdx === toIdx) return;
      const rect = row.getBoundingClientRect();
      const dropBelow = (e.clientY - rect.top) > rect.height / 2;
      settingsReorderComingSoon(fromIdx, toIdx, dropBelow);
    });
  });
}
function settingsReorderComingSoon(fromIdx, targetIdx, placeBelow) {
  settingsEnsureSchema();
  const list = S.raw.coming_soon;
  if (fromIdx < 0 || fromIdx >= list.length) return;
  if (typeof snapshot === 'function') snapshot();

  const [moving] = list.splice(fromIdx, 1);
  // Compute fresh target index after splice.
  let insertAt = targetIdx;
  if (fromIdx < targetIdx) insertAt -= 1;
  if (placeBelow) insertAt += 1;
  if (insertAt < 0) insertAt = 0;
  if (insertAt > list.length) insertAt = list.length;
  list.splice(insertAt, 0, moving);

  S.dirty.add('__manifest__');
  updateDirtyBadge();
  settingsRenderComingSoon();
  toast('Reordered coming-soon list', 'ok');
}

function settingsUpdateAnnouncementPreview() {
  const a = S.raw.announcement || {};
  const ribbon = document.getElementById('settings-anno-preview');
  const textEl = document.getElementById('settings-anno-preview-text');
  const ctaEl  = document.getElementById('settings-anno-preview-cta');
  if (!ribbon || !textEl || !ctaEl) return;

  const style = SETTINGS_VALID_STYLES.includes(a.style) ? a.style : 'neon';
  ribbon.classList.toggle('is-alert', style === 'alert');
  ribbon.classList.toggle('is-info',  style === 'info');
  ribbon.classList.toggle('is-disabled', !a.enabled);

  const text = (a.text || '').trim();
  textEl.textContent = text || 'Announcement text will appear here.';
  textEl.style.opacity = text ? '1' : '0.55';

  if (a.link && a.link_label) {
    ctaEl.style.display = '';
    ctaEl.textContent = a.link_label;
    // Make sure it's not actually navigable from the preview.
    ctaEl.href = 'javascript:void(0)';
    ctaEl.title = a.link;
  } else {
    ctaEl.style.display = 'none';
  }
}
