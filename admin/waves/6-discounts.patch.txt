  document.getElementById('grid').style.display = 'none';

  document.getElementById('discounts-body').style.display = hasSecret ? '' : 'none';
  if (hasSecret) {
    populateDiscountProductPicker();
    loadDiscounts();
  }
}

// -------- UI helpers --------
function updateDiscountTypeUI() {
  const type = document.querySelector('input[name="dc-type"]:checked').value;
  const amountLabel = document.getElementById('dc-amount-label');
  const amountHint = document.getElementById('dc-amount-hint');
  const amountInp = document.getElementById('dc-amount');
  const ccy = document.getElementById('dc-currency');
  if (type === 'percentage') {
    amountLabel.textContent = 'Amount (%)';
    amountHint.textContent = '1–100 (percent)';
    amountInp.min = '1'; amountInp.max = '100'; amountInp.placeholder = '30';
    ccy.disabled = true;
  } else {
    amountLabel.textContent = 'Amount (cents)';
    amountHint.textContent = 'Cents — e.g. 500 = $5.00 off';
    amountInp.min = '1'; amountInp.removeAttribute('max'); amountInp.placeholder = '500';
    ccy.disabled = false;
  }
}

function updateDiscountScopeUI() {
  const scope = document.getElementById('dc-scope').value;
  document.getElementById('dc-products-wrap').style.display = scope === 'specific' ? '' : 'none';
}

function populateDiscountProductPicker() {
  // Fill the <select multiple> from currently-loaded catalog (S.products).
  // Paddle's product_id lives on each p.paddle.product_id — fall back to p.id when absent.
  const sel = document.getElementById('dc-products');
  if (!sel) return;
  const opts = [];
  const list = (typeof S !== 'undefined' && Array.isArray(S.products)) ? S.products : [];
  for (const p of list) {
    const pid = (p && p.paddle && p.paddle.product_id) ? p.paddle.product_id : null;
    if (!pid) continue; // only products actually mapped to a Paddle product
    opts.push('<option value="' + escapeHtml(pid) + '">' + escapeHtml(p.name || p.id) + ' — ' + escapeHtml(pid) + '</option>');
  }
  sel.innerHTML = opts.length
    ? opts.join('')
    : '<option disabled>No products with paddle.product_id set in products.json</option>';
}

function resetDiscountForm() {
  document.getElementById('dc-code').value = '';
  document.getElementById('dc-amount').value = '';
  document.getElementById('dc-desc').value = '';
  document.getElementById('dc-starts').value = '';
  document.getElementById('dc-expires').value = '';
  document.getElementById('dc-usage-limit').value = '';
  document.getElementById('dc-scope').value = 'all';

  const psel = document.getElementById('dc-products');
  if (psel) [...psel.options].forEach(o => (o.selected = false));
  updateDiscountTypeUI();
  updateDiscountScopeUI();

}

function showDiscountBanner(msg, kind) {
  const b = document.getElementById('discounts-banner');
  if (!b) return;
  if (!msg) { b.style.display = 'none'; b.textContent = ''; b.classList.remove('ok'); return; }
  b.textContent = msg;
  b.classList.toggle('ok', kind === 'ok');
  b.style.display = '';
}

// -------- Create --------
async function createDiscount() {
  const secret = getSalesSecret();
  if (!secret) { showDiscountBanner('Admin secret missing — set it up in Sales view first.', 'err'); return; }

  const code = (document.getElementById('dc-code').value || '').trim();
  const type = document.querySelector('input[name="dc-type"]:checked').value;
  const amountRaw = (document.getElementById('dc-amount').value || '').trim();
  const desc = (document.getElementById('dc-desc').value || '').trim();
  const starts = (document.getElementById('dc-starts').value || '').trim();
  const expires = (document.getElementById('dc-expires').value || '').trim();
  const usageLimitRaw = (document.getElementById('dc-usage-limit').value || '').trim();
  const scope = document.getElementById('dc-scope').value;
  const ccy = document.getElementById('dc-currency').value;

  // Validate
  if (code.length < 3 || code.length > 40) { showDiscountBanner('Code must be 3–40 characters.', 'err'); return; }
  if (!amountRaw) { showDiscountBanner('Amount is required.', 'err'); return; }
  const amountNum = Number(amountRaw);
  if (!isFinite(amountNum) || amountNum <= 0) { showDiscountBanner('Amount must be a positive number.', 'err'); return; }
  if (type === 'percentage' && amountNum > 100) { showDiscountBanner('Percent amount cannot exceed 100.', 'err'); return; }

  const body = {
    code,
    type,
    amount: String(amountNum),
    enabled_for_checkout: true,
  };
  if (desc) body.description = desc;
  if (type === 'flat') body.currency_code = ccy;
  if (starts) body.starts_at = new Date(starts).toISOString();
  if (expires) body.expires_at = new Date(expires).toISOString();
  if (usageLimitRaw) {
    const u = parseInt(usageLimitRaw, 10);
    if (isFinite(u) && u > 0) body.usage_limit = u;
  }
  if (scope === 'specific') {
    const sel = document.getElementById('dc-products');
    const pids = [...sel.selectedOptions].map(o => o.value).filter(v => !!v);
    if (!pids.length) { showDiscountBanner('Pick at least one product, or change scope to "All products".', 'err'); return; }
    body.restrict_to = pids;
  }

  const btn = document.getElementById('dc-create-btn');
  btn.disabled = true; btn.textContent = 'Creating...';
  showDiscountBanner('', null);

  try {
    const r = await fetch('/api/paddle/discounts', {
      method: 'POST',
      headers: {
        'x-admin-secret': secret,
        'Content-Type': 'application/json',
        'Accept': 'application/json',
      },
      body: JSON.stringify(body),
    });
    const txt = await r.text();
    let j = null; try { j = JSON.parse(txt); } catch (_) {}
    if (!r.ok) {
      const errMsg = (j && j.error && (j.error.detail || j.error.message)) || txt.slice(0, 300);
      throw new Error('Paddle ' + r.status + ': ' + errMsg);
    }
    showDiscountBanner('Created discount ' + code + '.', 'ok');
    toast('Discount ' + code + ' created', 'ok');
    resetDiscountForm();
    loadDiscounts();
  } catch (e) {
    showDiscountBanner('Create failed: ' + (e.message || e), 'err');
    toast('Create failed', 'err');
  } finally {
    btn.disabled = false; btn.textContent = '+ Create discount';
  }
}

// -------- List --------
async function loadDiscounts() {
  const secret = getSalesSecret();
  if (!secret) { showDiscountsView(); return; }
  const btn = document.getElementById('discounts-refresh-btn');
  btn.disabled = true; btn.textContent = '⟳ Loading...';
  showDiscountBanner('', null);

  try {
    // Paddle paginates; grab every discount (active + archived + expired) in one call set.
    const all = [];
    let url = '/api/paddle/discounts?per_page=100';
    const MAX_PAGES = 20;
    let page = 0;
    while (url && page < MAX_PAGES) {
      page++;
      const r = await fetch(url, { headers: { 'x-admin-secret': secret, 'Accept': 'application/json' } });
      if (r.status === 401) throw new Error('admin secret rejected — reset in Sales view');
      if (r.status === 403) throw new Error('proxy refused /discounts — worker whitelist may be out of date');
      if (!r.ok) throw new Error('Paddle ' + r.status + ': ' + (await r.text()).slice(0, 200));
      const j = await r.json();
      const chunk = Array.isArray(j.data) ? j.data : [];
      all.push(...chunk);
      btn.textContent = '⟳ Loading... (' + all.length + ')';
      const pag = j.meta && j.meta.pagination;
      const hasMore = pag && pag.has_more === true;
      if (!hasMore || chunk.length === 0 || !pag.next) { url = null; break; }
      try {
        const parsed = new URL(pag.next);
        url = '/api/paddle' + parsed.pathname + parsed.search;
      } catch (_) { url = null; }
    }

    DISCOUNTS_STATE.all = all;
    renderDiscountsTable();
    document.getElementById('discounts-last-sync').textContent = 'Last synced ' + new Date().toLocaleString();
    const badge = document.getElementById('c-discounts');
    if (badge) badge.textContent = all.filter(d => d.status === 'active').length;
  } catch (e) {
    showDiscountBanner('Load failed: ' + (e.message || e), 'err');
    document.getElementById('discounts-active-rows').innerHTML =
      '<tr><td colspan="7" class="discounts-empty" style="color:var(--danger)">Failed: ' + escapeHtml(e.message || String(e)) + '</td></tr>';
  } finally {
    btn.disabled = false; btn.textContent = '⟳ Refresh';
  }
}

// -------- Render --------
function renderDiscountsTable() {
  const all = DISCOUNTS_STATE.all || [];
  const now = Date.now();
  const isExpired = (d) => d.expires_at && new Date(d.expires_at).getTime() < now;

  const active = all.filter(d => d.status === 'active' && !isExpired(d));
  const archived = all.filter(d => d.status !== 'active' || isExpired(d));

  const activeEl = document.getElementById('discounts-active-rows');
  const archivedEl = document.getElementById('discounts-archived-rows');

  document.getElementById('discounts-active-count').textContent = '· ' + active.length;
  document.getElementById('discounts-archived-count').textContent = '· ' + archived.length;

  activeEl.innerHTML = active.length
    ? active.map(d => renderDiscountRow(d, false)).join('')
    : '<tr><td colspan="7" class="discounts-empty">No active discounts.</td></tr>';

  archivedEl.innerHTML = archived.length
    ? archived.map(d => renderDiscountRow(d, true)).join('')
    : '<tr><td colspan="7" class="discounts-empty">—</td></tr>';
}

function renderDiscountRow(d, isArchived) {
  const expired = d.expires_at && new Date(d.expires_at).getTime() < Date.now();
  const statusLabel = expired && d.status === 'active' ? 'expired' : (d.status || '—');
  const statusClass = (statusLabel === 'active') ? 'active'
    : (statusLabel === 'expired') ? 'expired'
    : 'archived';

  const amount = d.type === 'percentage'
    ? (d.amount + '%')
    : fmtCurrencyAmount(d.amount, d.currency_code);

  const uses = (d.times_used != null ? d.times_used : 0) + (d.usage_limit ? ' / ' + d.usage_limit : '');

  const scope = (Array.isArray(d.restrict_to) && d.restrict_to.length)
    ? d.restrict_to.length + ' product' + (d.restrict_to.length === 1 ? '' : 's')
    : 'All products';

  const expiresCol = d.expires_at ? fmtDate(d.expires_at) : (isArchived ? '—' : 'Never');

  const actionCell = isArchived
    ? '<span class="dc-status ' + escapeHtml(statusClass) + '">' + escapeHtml(statusLabel) + '</span>'
    : '<button class="dc-deactivate" onclick="event.stopPropagation();deactivateDiscount(\'' + escapeHtml(d.id) + '\', \'' + escapeHtml(d.code || '') + '\')">Deactivate</button>';

  const mainRow = '<tr class="dc-row-main" onclick="toggleDiscountDetail(\'' + escapeHtml(d.id) + '\')">' +
    '<td><span class="dc-code">' + escapeHtml(d.code || d.id) + '</span></td>' +
    '<td><span class="dc-type ' + escapeHtml(d.type || '') + '">' + escapeHtml(d.type || '—') + '</span></td>' +
    '<td class="num">' + escapeHtml(amount) + '</td>' +
    '<td>' + escapeHtml(expiresCol) + '</td>' +
    '<td class="num">' + escapeHtml(uses) + '</td>' +
    (isArchived
      ? '<td>' + escapeHtml(statusLabel) + '</td>'
      : '<td>' + escapeHtml(scope) + '</td>') +
    '<td class="actions">' + actionCell + '</td>' +
  '</tr>';

  if (DISCOUNTS_STATE.expandedId !== d.id) return mainRow;

  // Detail row
  const restrictStr = (Array.isArray(d.restrict_to) && d.restrict_to.length)
    ? d.restrict_to.map(escapeHtml).join(', ')
    : 'All products';
  const detail =
    '<tr class="dc-row-detail"><td colspan="7">' +
      '<div class="dc-detail-block">' +
        '<div>ID</div><div><b>' + escapeHtml(d.id) + '</b></div>' +
        '<div>Description</div><div>' + escapeHtml(d.description || '—') + '</div>' +
        '<div>Starts at</div><div>' + escapeHtml(d.starts_at ? fmtDate(d.starts_at) : '—') + '</div>' +
        '<div>Expires at</div><div>' + escapeHtml(d.expires_at ? fmtDate(d.expires_at) : 'Never') + '</div>' +
        '<div>Usage limit</div><div>' + (d.usage_limit != null ? escapeHtml(String(d.usage_limit)) : 'Unlimited') + '</div>' +
        '<div>Times used</div><div>' + escapeHtml(String(d.times_used != null ? d.times_used : 0)) + '</div>' +
        '<div>Enabled for checkout</div><div>' + (d.enabled_for_checkout ? 'Yes' : 'No') + '</div>' +
        '<div>Restricted to</div><div class="dc-scope-list">' + restrictStr + '</div>' +
        '<div>Created</div><div>' + escapeHtml(d.created_at ? fmtDate(d.created_at) : '—') + '</div>' +
      '</div>' +
    '</td></tr>';

  return mainRow + detail;
}

function fmtCurrencyAmount(cents, ccy) {
  const n = Number(cents) / 100;
  if (!isFinite(n)) return String(cents);
  try {
    return new Intl.NumberFormat(undefined, { style: 'currency', currency: ccy || 'USD' }).format(n);
  } catch (_) {
    return (ccy || 'USD') + ' ' + n.toFixed(2);
  }
}

function toggleDiscountDetail(id) {
  DISCOUNTS_STATE.expandedId = (DISCOUNTS_STATE.expandedId === id) ? null : id;
  renderDiscountsTable();
}

function toggleArchivedDiscounts() {
  DISCOUNTS_STATE.archivedOpen = !DISCOUNTS_STATE.archivedOpen;
  document.getElementById('discounts-archived-wrap').style.display = DISCOUNTS_STATE.archivedOpen ? '' : 'none';
  document.getElementById('discounts-archived-caret').innerHTML = DISCOUNTS_STATE.archivedOpen ? '&#9662;' : '&#9656;';
}

// -------- Deactivate --------
async function deactivateDiscount(id, code) {
  if (!confirm('Deactivate discount ' + (code || id) + '? Customers will no longer be able to use it.')) return;
  const secret = getSalesSecret();
  if (!secret) { showDiscountBanner('Admin secret missing.', 'err'); return; }

  try {
    const r = await fetch('/api/paddle/discounts/' + encodeURIComponent(id), {
      method: 'PATCH',
      headers: {
        'x-admin-secret': secret,
        'Content-Type': 'application/json',
        'Accept': 'application/json',
      },
      body: JSON.stringify({ status: 'archived' }),
    });
    const txt = await r.text();
    let j = null; try { j = JSON.parse(txt); } catch (_) {}
    if (!r.ok) {
      const errMsg = (j && j.error && (j.error.detail || j.error.message)) || txt.slice(0, 200);
      throw new Error('Paddle ' + r.status + ': ' + errMsg);
    }
    toast('Deactivated ' + (code || id), 'ok');
    showDiscountBanner('Deactivated ' + (code || id) + '.', 'ok');
    loadDiscounts();
  } catch (e) {
    showDiscountBanner('Deactivate failed: ' + (e.message || e), 'err');
    toast('Deactivate failed', 'err');
  }
}
