WAVE 3 — HOMEPAGE EDITOR
========================
Adds a Homepage view to the editor so the user can edit hero copy,
tagline strip, HUD ticker, featured section headings, and the footer
tiles/CTAs without touching HTML. All edits land in
S.raw.homepage and mark __manifest__ dirty so Save writes products.json.

Insertion points (editor.html):
  NAV    -> inside the Site nav section (near data-view="settings")
  PANEL  -> inside the canvas body, next to #sales-panel / #featured-panel
  CSS    -> bottom of <style> block (all selectors .homepage-* prefixed)
  JS     -> end of <script> block, next to showFeaturedView()
  Also add to onViewChange-style dispatcher:
     if (view === 'homepage') showHomepageView();

/* ============================================================
 * Wave 3 — Homepage editor view
 * ============================================================ */

function ensureHomepageObject() {
  if (!S.raw) S.raw = {};
  if (!S.raw.homepage || typeof S.raw.homepage !== 'object') {
    S.raw.homepage = {};
  }
  var hp = S.raw.homepage;
  if (!hp.hero || typeof hp.hero !== 'object') hp.hero = {};
  if (!Array.isArray(hp.tagline_strip)) hp.tagline_strip = [];
  if (!Array.isArray(hp.ticker_strip)) hp.ticker_strip = [];
  if (!hp.sections || typeof hp.sections !== 'object') hp.sections = {};
  if (!hp.footer || typeof hp.footer !== 'object') hp.footer = {};
  return hp;
}

/* Dotted path get/set on an object, supporting numeric indices ("a.0"). */
function getHomepagePath(path) {
  var hp = ensureHomepageObject();
  var parts = String(path).split('.');
  var cur = hp;
  for (var i = 0; i < parts.length; i++) {
    if (cur == null) return undefined;
    cur = cur[parts[i]];
  }
  return cur;
}

function setHomepagePath(path, val) {
  var hp = ensureHomepageObject();
  var parts = String(path).split('.');
  var cur = hp;
  for (var i = 0; i < parts.length - 1; i++) {
    var k = parts[i];
    var next = parts[i + 1];
    var nextIsIndex = /^\d+$/.test(next);
    if (cur[k] == null) {
      cur[k] = nextIsIndex ? [] : {};
    }
    cur = cur[k];
  }
  var leaf = parts[parts.length - 1];
  cur[leaf] = val;
}

function showHomepageView() {
  document.getElementById('grid').style.display = 'none';
  var fp = document.getElementById('featured-panel');
  if (fp) fp.style.display = 'none';
  var sp = document.getElementById('sales-panel');
  if (sp) sp.style.display = 'none';
  document.getElementById('homepage-panel').style.display = '';
  document.getElementById('canvas-title').textContent = 'Homepage';
  document.getElementById('canvas-sub').textContent = 'Hero, tagline, ticker, section heads, footer tiles';
  renderHomepagePanel();
}

function renderHomepagePanel() {
  ensureHomepageObject();
  var panel = document.getElementById('homepage-panel');
  if (!panel) return;

  // Fill every [data-hp-key] input/textarea with its current value.
  panel.querySelectorAll('[data-hp-key]').forEach(function (el) {
    var key = el.getAttribute('data-hp-key');
    var val = getHomepagePath(key);
    el.value = (val == null) ? '' : String(val);

    // Wire change events once.
    if (!el.__hpBound) {
      el.__hpBound = true;
      el.addEventListener('input', onHomepageFieldChange);
    }
  });

  renderTickerList();
  updateHomepagePreview();
}

function onHomepageFieldChange(ev) {
  var el = ev.currentTarget;
  var key = el.getAttribute('data-hp-key');
  if (!key) return;
  var v = el.value;
  // Strip trailing empty -> delete path so fallback wins.
  if (v === '') {
    setHomepagePath(key, '');
  } else {
    setHomepagePath(key, v);
  }
  S.dirty.add('__manifest__');
  updateDirtyBadge && updateDirtyBadge();
  updateHomepagePreview();
}

/* ---- Ticker dynamic list ---- */
function renderTickerList() {
  var hp = ensureHomepageObject();
  var list = document.getElementById('homepage-ticker-list');
  if (!list) return;
  list.innerHTML = '';
  hp.ticker_strip.forEach(function (item, i) {
    var row = document.createElement('div');
    row.className = 'homepage-ticker-row';
    row.innerHTML =
      '<div class="idx">' + (i + 1) + '.</div>' +
      '<input type="text" value="" />' +
      '<button class="rm" type="button" title="Remove">&times;</button>';
    var input = row.querySelector('input');
    input.value = item == null ? '' : String(item);
    input.addEventListener('input', function () {
      var hp2 = ensureHomepageObject();
      hp2.ticker_strip[i] = input.value;
      S.dirty.add('__manifest__');
      updateDirtyBadge && updateDirtyBadge();
    });
    row.querySelector('.rm').addEventListener('click', function () {
      var hp2 = ensureHomepageObject();
      hp2.ticker_strip.splice(i, 1);
      S.dirty.add('__manifest__');
      updateDirtyBadge && updateDirtyBadge();
      renderTickerList();
    });
    list.appendChild(row);
  });
}

function addTickerItem() {
  var hp = ensureHomepageObject();
  hp.ticker_strip.push('');
  S.dirty.add('__manifest__');
  updateDirtyBadge && updateDirtyBadge();
  renderTickerList();
}

/* ---- Live preview ---- */
function updateHomepagePreview() {
  var hp = ensureHomepageObject();
  var $ = function (id) { return document.getElementById(id); };

  if ($('hp-prev-crafted')) $('hp-prev-crafted').textContent = hp.hero.crafted || 'CRAFTED';
  if ($('hp-prev-for'))     $('hp-prev-for').textContent     = hp.hero.for     || 'FOR';
  if ($('hp-prev-misfits')) $('hp-prev-misfits').textContent = hp.hero.misfits || 'MISFITS.';
  if ($('hp-prev-sub'))     $('hp-prev-sub').textContent     = hp.hero.subtitle || '48 weaponized Rust plugins...';

  if ($('hp-prev-eyebrow')) {
    var t = hp.tagline_strip || [];
    var a = t[0] || '>> ONE DEV';
    var b = t[1] || 'EVERY LINE';
    var c = t[2] || 'ZERO FILLER <<';
    $('hp-prev-eyebrow').textContent = a + '  ·  ' + b + '  ·  ' + c;
  }
}
